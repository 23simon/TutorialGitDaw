apt-get install git => instalación de git
add-apt-repository ppa:git-core/ppa # apt update; apt install git => actualización a última versión
git --version => comprobar version de Git
git init => crear repositorio Git
git config --list => ver configuración de Git
git status => comprobar cambios non actualizados
git add . => actualizar cambios
git commit -m “comentario” => comentar os cambios realizados
git branch -M main => mover rama master a unha chamada main
git remote add etiqueta URL_do_repositorio_GitHub => relacionar URL con unha etiqueta
